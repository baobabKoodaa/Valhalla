1   package gui;

3   import javax.imageio.ImageIO;
4   import java.awt.*;
5   import java.io.File;
6   import java.io.IOException;
7   
8   /**
9    * Button has a (pre-loaded) image and an identifier (name).
10   */
11  public class Button {
12      /** Name acts as an identifier. */
13      String name;
14  
15      /** Pre-loaded image file. */
16      Image img;
17  
18      /** Coordinates on screen. */
19      int y;
20      int x;
21  
22      /** Size of the button. */
23      int width;
24      int height;
25  
26      /** Constructor initializes button.
27       *
28       * @param name name
29       * @param x x
30       * @param y y
31       * @param width width
32       * @param height height this is stupid
33       */
34      public Button(String name, int x, int y, int width, int height) {
35          this.name = name;
36          this.x = x;
37          this.y = y;
38          this.width = width;
39          this.height = height;
40          boolean oneUp = true; /* Image path varies depending if we run from JAR or IDE. */
41          boolean success = tryLoadingImagePath(oneUp);
42          if (!success) tryLoadingImagePath(!oneUp);
43          /* If we still fail, continue without images. */
44      }
45  
46      private boolean tryLoadingImagePath(boolean oneUp) {
47          String s = File.separator; /* To work on linux and windows etc. */
48          String relativePath = "Valhalla" + s + "src" + s + "resources" + s + name + ".png";
49          if (oneUp) relativePath = ".." + s + relativePath;
50          try {
51              this.img = ImageIO.read(new File(relativePath));
52          } catch (IOException e) {
53              System.out.println("Unable to preload image " + relativePath);
54              return false;
55          }
56          return true;
57      }
58  
59      /** Draws button. */
60      public void draw(Graphics2D g2d) {
61          if (img == null) {
62              /* If preloading image failed earlier, don't crash the program */
63              return;
64          }
65          g2d.drawImage(img, x, y, width, height, null);
66      }
67  }

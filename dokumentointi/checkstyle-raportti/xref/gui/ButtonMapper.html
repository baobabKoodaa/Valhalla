5   import java.util.ArrayList;
6   import java.util.HashMap;
7   import java.util.List;
8   
9   public class ButtonMapper {
10      List<Button> iterableList;
11      HashMap<Pair, Button> pixelMap;
12  
13      public ButtonMapper() {
14          iterableList = new ArrayList<>();
15          pixelMap = new HashMap<>();
16      }
17  
18      public void add(Button button) {
19          iterableList.add(button);
20          /* Add all pixels of button to pixelMap */
21          int minY = button.y;
22          int maxY = minY + button.height;
23          int minX = button.x;
24          int maxX = minX + button.width;
25          for (int y = minY; y <= maxY; y++) {
26              for (int x = minX; x <= maxX; x++) {
27                  Pair point = new Pair(y, x);
28                  Button prev = pixelMap.put(point, button);
29                  if (prev != null) {
30                      System.out.println("Warning! Overlayed buttons " + prev.name + " and " + button.name);
31                  }
32              }
33          }
34      }
35  
36      public Button getButton(Pair point) {
37          return pixelMap.get(point);
38      }
39  
40      public List<Button> getList() {
41          return iterableList;
42      }
43  }

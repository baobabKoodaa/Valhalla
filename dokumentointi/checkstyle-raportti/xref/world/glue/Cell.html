8   import java.util.ArrayList;
9   import java.util.List;
10  
11  import static util.MagicNumbers.*;
12  
13  public class Cell {
14      public int y;
15      public int x;
16      private Terrain terrain;
17      private boolean[] visible;
18      private List<Element> elements;
19  
20      public Cell(int y, int x, Terrain terrain) {
21          this.y = y;
22          this.x = x;
23          this.terrain = terrain;
24          this.visible = new boolean[MAX_PLAYERS + 1];
25          this.elements = new ArrayList<>();
26      }
27  
28      public void addElement(Element elem) {
29          elements.add(elem);
30      }
31  
32      public Element getTopElement() {
33          if (isEmpty()) {
34              return null;
35          }
36          return elements.get(elements.size() - 1);
37      }
38  
39      public void removeTopElement() {
40          elements.remove(elements.size() - 1);
41      }
42  
43      public boolean isEmpty() {
44          return (elements.isEmpty());
45      }
46  
47      public Terrain getTerrain() {
48          return terrain;
49      }
50  
51      public boolean isVisibleTo(int player) {
52          return visible[player];
53      }
54  
55      public void setVisibleTo(int player) {
56          visible[player] = true;
57      }
58  
59      /* Used in map generation to create clusters of food */
60      public boolean hasFood() {
61          for (Element element : elements) {
62              if (element.canBeEatenBy(HUMAN_PLAYER)) {
63                  return true;
64              }
65          }
66          return false;
67      }
68  
69      /* Returns true if food was available. Only eats 1 food. */
70      public boolean eatFoodAs(int player) {
71          for (int i = elements.size() - 1; i >= 0; i--) {
72              Element element = elements.get(i);
73              if (element.canBeEatenBy(player)) {
74                  elements.remove(i);
75                  return true;
76              }
77          }
78          return false;
79      }
80  
81      public boolean hasLimbOf(int player) {
82          if (isEmpty()) {
83              return false;
84          }
85          if (getTopElement().getClass() != Limb.class) {
86              return false;
87          }
88          Limb limb = (Limb) getTopElement();
89          Organism o = limb.getOrganism();
90          return (o.getPlayer() == player);
91      }
92  
93  
94  }

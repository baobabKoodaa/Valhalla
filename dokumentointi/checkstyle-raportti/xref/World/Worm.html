7   public class Worm extends Organism {
8       Deque<Cell> partsOfWorm;
9       int length;
10  
11      @Override
12      public void physicalBirth(Cell birthPlace) {
13          this.length = 4;
14          this.partsOfWorm = new ArrayDeque<>();
15          /* x,y denotes head */
16          this.y = birthPlace.y;
17          this.x = birthPlace.x;
18          Limb head = new Limb(this);
19          head.setAsHead();
20          birthPlace.addElement(head);
21          partsOfWorm.addLast(birthPlace);
22          clearLineOfSight();
23      }
24  
25      @Override
26      public void takeDamage(Cell cell) {
27          dropTailUpTo(cell);
28          if (partsOfWorm.isEmpty()) {
29              this.alive = false;
30          }
31      }
32  
33      @Override
34      public void live(Cell[][] map) {
35          Cell prev = partsOfWorm.pollLast(); /* We will add this back */
36          Element prevHead = map[y][x].getTopElement();
37          prevHead.unsetAsHead();
38          Cell moveTo = getState().getRandomAdjacentNonFriendlyCell(y, x, getPlayer());
39          if (moveTo == null) {
40              moveTo = getSomeCannibalizingMove();
41              Limb limb = (Limb) moveTo.getTopElement();
42              if (limb.organism == this) {
43                  dropTailUpTo(moveTo);
44              } else {
45                  System.out.println("Killed a friendly. Check this works ok.");
46                  limb.organism.takeDamage(moveTo);
47              }
48          }
49          y = moveTo.y;
50          x = moveTo.x;
51          partsOfWorm.addLast(prev);
52          partsOfWorm.addLast(moveTo);
53          if (map[y][x].eatFoodAs(getPlayer())) {
54              length++;
55          }
56          if (partsOfWorm.size() > length) {
57              Cell tail = partsOfWorm.pollFirst();
58              map[tail.y][tail.x].removeTopElement();
59          }
60          Limb head = new Limb(this);
61          head.setAsHead();
62          map[y][x].addElement(head);
63          clearLineOfSight();
64      }
65  
66      private Cell getSomeCannibalizingMove() {
67          List<Cell> neighbors = getState().getNeighboursWithinRadius(y, x, 1);
68          Collections.shuffle(neighbors);
69          for (Cell neighbor : neighbors) {
70              if (neighbor.y != y && neighbor.x != x) {
71                  continue; /* Diagonals are not ok */
72              }
73              /* Head has been temporarily polled from partsOfWorm to reveal neck */
74              if (!partsOfWorm.isEmpty() && neighbor == partsOfWorm.peekLast()) {
75                  /* To stop worms from doing 180Â¤ */
76                  continue;
77              }
78              return neighbor;
79          }
80          return null; /* Never returns null */
81      }
82  
83      private void dropTailUpTo(Cell eatenCell) {
84          for (int i = partsOfWorm.size() - 1; i >= 0; i--) {
85              Cell tail = partsOfWorm.pollFirst();
86              length--;
87              tail.removeTopElement();
88              tail.addElement(new Remains(getPlayer()));
89              if (tail == eatenCell) {
90                  break;
91              }
92          }
93      }
94  
95      private void clearLineOfSight() {
96          List<Cell> cellsInRange = getState().getCellsWithinRadius(y, x, NANOBOT_LINE_OF_SIGHT);
97          for (Cell cell : cellsInRange) {
98              cell.setVisibleTo(getPlayer());
99          }
100     }
101 }

14  public class Painter extends JPanel {
15      public boolean dontTouchThePaint;
16      private State state;
17      private int cellSideLength;
18      private double offsetY;
19      private double offsetX;
20      private int viewHeight;
21      private int viewWidth;
22  
23      private Pair selectedSq;
24  
25      public Painter(State state, int width, int height) {
26          this.state = state;
27          this.viewWidth = width;
28          this.viewHeight = height;
29          this.setPreferredSize(new Dimension(width, height));
30          this.setBackground(Color.white);
31          this.cellSideLength = 50;
32          InputListener inputListener = new InputListener(this);
33          addMouseListener(inputListener);
34          addMouseMotionListener(inputListener);
35          addMouseWheelListener(inputListener);
36          repaint();
37      }
38  
39      @Override
40      public void paintComponent(Graphics g) {
41          while (state.updateInProgress) {
42              tryToSleep(1L);
43          }
44          dontTouchThePaint = true; /* Should be set also everywhere that repaint is asked */
45          super.paintComponent(g);
46          Graphics2D g2d = (Graphics2D) g;
47  
48          Cell[][] map = state.getMap();
49          int topY = (int)offsetY/ cellSideLength;
50          int leftX = (int)offsetX/ cellSideLength;
51          int bottomY = 1 + topY + (int) Math.ceil(viewHeight / cellSideLength);
52          int rightX = 1 + leftX + (int) Math.ceil(viewWidth / cellSideLength);
53          while (bottomY >= map.length) bottomY--;
54          while (rightX >= map[bottomY].length) rightX--;
55  
56          int viewY = - (int)offsetY % cellSideLength;
57          for (int y=topY; y<=bottomY; y++, viewY+=cellSideLength) {
58              int viewX = - (int)offsetX % cellSideLength;
59              for (int x=leftX; x<=rightX; x++, viewX+=cellSideLength) {
60                  Cell cell = map[y][x];
61                  if (cell.isVisibleTo(HUMAN_PLAYER)) drawCell(g2d, cell, viewY, viewX);
62                  else drawUndiscoveredArea(g2d, viewY, viewX);
63              }
64          }
65          /* Draw selection */
66          if (selectedSq != null) {
67              viewY = - (int)offsetY % cellSideLength + (selectedSq.y - topY) * cellSideLength;
68              int viewX = - (int)offsetX % cellSideLength + (selectedSq.x - leftX) * cellSideLength;
69              g2d.setColor(Color.YELLOW);
70              g2d.drawRect(viewX, viewY, cellSideLength, cellSideLength);
71          }
72          dontTouchThePaint = false; /* Concurrency related flag */
73      }
74  
75      public void drawCell(Graphics2D g2d, Cell cell, int viewY, int viewX) {
76          /* Draw terrain */
77          g2d.setColor(getColorForTerrain(cell.getTerrain()));
78          g2d.fillRect(viewX, viewY, cellSideLength, cellSideLength);
79          /* Draw element */
80          if (cell.getTopElement() != null) {
81              g2d.setColor(cell.getTopElement().getColor());
82              g2d.fillRect(viewX, viewY, cellSideLength, cellSideLength);
83              if (cell.getTopElement().paintAsHead()) {
84                  g2d.setColor(HEAD_MARKER_COLOR);
85                  int halvedSide = cellSideLength/2;
86                  int x = viewX + cellSideLength/4;
87                  int y = viewY + cellSideLength/4;
88                  g2d.fillRect(x, y, halvedSide, halvedSide);
89              }
90          }
91          /* Draw grid */
92          g2d.setColor(Color.LIGHT_GRAY);
93          g2d.drawRect(viewX, viewY, cellSideLength, cellSideLength);
94      }
95  
96      public void drawUndiscoveredArea(Graphics2D g2d, int viewY, int viewX) {
97          g2d.setColor(Color.BLACK);
98          g2d.fillRect(viewX, viewY, cellSideLength, cellSideLength);
99      }

101     public void userClickedOn(Pair point) {
102         int mapX = getMapXFromView(point.x);
103         int mapY = getMapYFromView(point.y);
104         if (selectedSq != null) selectedSq = null;
105         else selectedSq = new Pair(mapY, mapX);
106         repaint();
107     }
108 
109     public void zoomIn() {
110         int newZoom = (int) Math.round(1.1 * this.cellSideLength);
111         applyZoom(newZoom);
112     }
113 
114     public void zoomOut() {
115         if (this.cellSideLength * state.getMap().length < viewHeight) return;
116         int newZoom = (int) Math.round(this.cellSideLength / 1.1);
117         applyZoom(newZoom);
118     }
119 
120     public void applyZoom(int newZoom) {
121         int prevZoom = this.cellSideLength;
122         this.cellSideLength = newZoom;
123         double change = 1.0 * prevZoom / newZoom;
124         this.offsetY /= change;
125         this.offsetX /= change;
126         forceOffsetWithinBounds();
127         repaint();
128     }
129 
130 
131     public void mouseDragged(int dx, int dy) {
132         setOffset(offsetY + dy, offsetX + dx);
133     }
134 
135     public void setOffset(double y, double x) {
136         this.offsetY = y;
137         this.offsetX = x;
138         forceOffsetWithinBounds();
139         repaint();
140     }
141 
142     private void forceOffsetWithinBounds() {
143         this.offsetX = Math.max(0, this.offsetX);
144         this.offsetY = Math.max(0, this.offsetY);
145         while (tooMuchToTheRight() && this.offsetX > 0) this.offsetX--;
146         while (tooMuchToTheSouth() && this.offsetY > 0) this.offsetY--;
147     }
148 
149     private boolean tooMuchToTheSouth() {
150         int y = getMapYFromView(viewHeight);
151         return (y >= state.getMap().length);
152     }
153 
154     private boolean tooMuchToTheRight() {
155         int x = getMapXFromView(viewWidth);
156         return (x >= state.getMap()[0].length);
157     }
158 
159     public int getMapXFromView(int viewX) {
160         return (int)(offsetX+viewX)/ cellSideLength;
161     }
162 
163     public int getMapYFromView(int viewY) {
164         return (int)(offsetY+viewY)/ cellSideLength;
165     }

167 }

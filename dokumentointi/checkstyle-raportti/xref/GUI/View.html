36      public View(State state) {
37          frame = new JFrame();
38          this.state = state;
39          this.viewWidth = 1200;

41          this.cellSideLength = 50;
42          canvas = new Canvas();

44          canvas.setBackground(Color.white);
45          InputListener inputListener = new InputListener(this);
46          canvas.addMouseListener(inputListener);

48          canvas.addMouseWheelListener(inputListener);

50          repaint();

67      private class Canvas extends JPanel {
68          @Override
69          public void paintComponent(Graphics g) {
70              while (state.updateInProgress) {
71                  tryToSleep(1L);
72              }
73              dontTouchThePaint = true; /* Because of concurrency, should be also set everywhere that repaint is asked */
74              Graphics2D g2d = (Graphics2D) g;
75              //System.out.println("GFX ? " + (g2d != null));
76  
77              Cell[][] map = state.getMap();
78              int topY = (int)offsetY/ cellSideLength;
79              int leftX = (int)offsetX/ cellSideLength;
80              int bottomY = 1 + topY + (int) Math.ceil(viewHeight / cellSideLength);
81              int rightX = 1 + leftX + (int) Math.ceil(viewWidth / cellSideLength);
82              while (bottomY >= map.length) bottomY--;
83              while (rightX >= map[bottomY].length) rightX--;
84  
85              int viewY = - (int)offsetY % cellSideLength;
86              for (int y=topY; y<=bottomY; y++, viewY+=cellSideLength) {
87                  int viewX = - (int)offsetX % cellSideLength;
88                  for (int x=leftX; x<=rightX; x++, viewX+=cellSideLength) {
89                      Cell cell = map[y][x];
90                      if (cell.isVisibleTo(HUMAN_PLAYER)) drawCell(g2d, cell, viewY, viewX);
91                      else drawUndiscoveredArea(g2d, viewY, viewX);
92                  }
93              }
94              /* Draw selection */
95              if (selectedSq != null) {
96                  viewY = - (int)offsetY % cellSideLength + (selectedSq.y - topY) * cellSideLength;
97                  int viewX = - (int)offsetX % cellSideLength + (selectedSq.x - leftX) * cellSideLength;
98                  g2d.setColor(Color.YELLOW);
99                  g2d.drawRect(viewX, viewY, cellSideLength, cellSideLength);
100             }
101             /* Draw media buttons */
102             for (Button button : buttonMapper.getList()) {
103                 button.draw(g2d);
104             }
105             g2d.setColor(Color.BLACK);
106             g2d.drawString(state.getRound() + "", viewWidth/2 + 63, 30);
107             dontTouchThePaint = false; /* Concurrency related flag */
108         }
109 
110         public void drawCell(Graphics2D g2d, Cell cell, int viewY, int viewX) {
111             /* Draw terrain */
112             g2d.setColor(getColorForTerrain(cell.getTerrain()));
113             g2d.fillRect(viewX, viewY, cellSideLength, cellSideLength);
114             /* Draw element */
115             if (cell.getTopElement() != null) {
116                 g2d.setColor(cell.getTopElement().getColor());
117                 g2d.fillRect(viewX, viewY, cellSideLength, cellSideLength);
118                 if (cell.getTopElement().paintAsHead()) {
119                     g2d.setColor(HEAD_MARKER_COLOR);
120                     int halvedSide = cellSideLength/2;
121                     int x = viewX + cellSideLength/4;
122                     int y = viewY + cellSideLength/4;
123                     g2d.fillRect(x, y, halvedSide, halvedSide);
124                 }
125             }
126             /* Draw grid */
127             g2d.setColor(Color.LIGHT_GRAY);
128             g2d.drawRect(viewX, viewY, cellSideLength, cellSideLength);
129         }
130 
131         public void drawUndiscoveredArea(Graphics2D g2d, int viewY, int viewX) {
132             g2d.setColor(Color.BLACK);
133             g2d.fillRect(viewX, viewY, cellSideLength, cellSideLength);
134         }
135     }

137     public void createButtons() {
138         buttonMapper = new ButtonMapper();
139         int buttonWidth = 32;
140         int buttonHeight = buttonWidth;
141         int buttonGap = buttonWidth/4;
142         int topYofButtons = 10;
143         int xOfFirstButton = viewWidth/2 - (3*buttonWidth+2*buttonGap)/2;
144         Button slower = new Button("slower", xOfFirstButton, topYofButtons, buttonWidth, buttonHeight);
145         Button pauseplay = new Button("pause", xOfFirstButton + buttonWidth + buttonGap, topYofButtons, buttonWidth, buttonHeight);
146         Button faster = new Button("faster", xOfFirstButton + 2*(buttonWidth + buttonGap), topYofButtons, buttonWidth, buttonHeight);
147         buttonMapper.add(slower);
148         buttonMapper.add(pauseplay);
149         buttonMapper.add(faster);
150     }

162     /* Interpret either as button click or selection of a cell */
163     public void userClickedOn(Pair point) {
164         Button button = buttonMapper.getButton(point);
165         if (button != null) {
166             pressButton(button.name);
167         }
168         else {
169             mapSelection(point);
170         }
171         repaint();
172     }

174     public void mapSelection(Pair point) {
175         int mapX = getMapXFromView(point.x);
176         int mapY = getMapYFromView(point.y);
177         if (selectedSq != null) {
178             selectedSq = null;
179         }
180         else {
181             selectedSq = new Pair(mapY, mapX);
182         }
183     }

185     public void pressButton(String name) {
186         switch (name) {
187             case "pause":
188                 gameLoop.pauseOrPlay();
189                 break;
190             case "faster":
191                 gameLoop.faster();
192                 break;
193             case "slower":

195                 break;
196             default:
197                 break;
198         }
199     }

201     public void zoomIn(Pair point) {
202         int newZoom = (int) Math.round(1.1 * this.cellSideLength);
203         applyZoom(newZoom, point);
204     }
205 
206     public void zoomOut(Pair point) {
207         if (this.cellSideLength * state.getMap().length < viewHeight) return;
208         int newZoom = (int) Math.round(this.cellSideLength / 1.1);
209         applyZoom(newZoom, point);
210     }
211 
212     public void applyZoom(int newZoom, Pair point) {
213         int prevZoom = this.cellSideLength;
214         this.cellSideLength = newZoom;
215         double change = 1.0 * prevZoom / newZoom;
216         this.offsetY /= change;
217         this.offsetX /= change;
218         this.offsetY += point.y/change;
219         this.offsetX += point.x/change;
220         this.offsetY -= point.y;
221         this.offsetX -= point.x;
222         forceOffsetWithinBounds();
223         repaint();
224     }

231     public void setOffset(double y, double x) {
232         this.offsetY = y;
233         this.offsetX = x;
234         forceOffsetWithinBounds();
235         repaint();
236     }
237 
238     private void forceOffsetWithinBounds() {
239         this.offsetX = Math.max(0, this.offsetX);
240         this.offsetY = Math.max(0, this.offsetY);
241         while (tooMuchToTheRight() && this.offsetX > 0) this.offsetX--;
242         while (tooMuchToTheSouth() && this.offsetY > 0) this.offsetY--;
243     }
244 
245     private boolean tooMuchToTheSouth() {
246         int y = getMapYFromView(viewHeight);
247         return (y >= state.getMap().length);
248     }
249 
250     private boolean tooMuchToTheRight() {
251         int x = getMapXFromView(viewWidth);
252         return (x >= state.getMap()[0].length);
253     }
254 
255     public int getMapXFromView(int viewX) {
256         return (int)(offsetX+viewX)/ cellSideLength;
257     }
258 
259     public int getMapYFromView(int viewY) {
260         return (int)(offsetY+viewY)/ cellSideLength;
261     }
